def my_norm(X, mn=[]):
    if len(mn) == 0:
        mn = np.mean(X,0).reshape(1,X.shape[1])
    nr = X - np.ones((X.shape[0],1))*mn
    return mn, nr

def my_scal(X, st=[]):
    if len(st) == 0:
        st = np.std(X,0).reshape(1,X.shape[1])
    sc = X / ( np.ones((X.shape[0],1))*st )
    return st, sc

def centering_my(X, *args):
    #
    if(len(args) > 0):
        X_mean = args[0]
    else:
        X_mean = np.mean(X, 0) # mean value of columns
    #
    Xc = X - np.ones((X.shape[0],1))*X_mean
    #
    return Xc, X_mean

def scaling_my(X, *args):
    #
    if ( len(args) > 0 ):
        X_std = args[0]
    else:
        X_std = np.std(X, axis=0) # Standard deviation of columns
    Xs = np.matrix(X) * np.matrix( np.linalg.inv(np.diag(X_std)) )
    #
    return Xs, X_std
